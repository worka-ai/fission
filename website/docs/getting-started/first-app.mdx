# Build your first app

Start by writing a minimal app that increments and decrements a number. Every part is visible.

```rust
use fission_core::prelude::*;
use fission_core::{Button, Column, Text, TextContent};

#[derive(Default, Debug, Clone, PartialEq, serde::Serialize, serde::Deserialize)]
struct CounterState {
    count: i32,
}

impl AppState for CounterState {}

#[derive(Action, serde::Serialize, serde::Deserialize, Debug, Clone, PartialEq, Eq)]
struct Increment;

#[derive(Action, serde::Serialize, serde::Deserialize, Debug, Clone, PartialEq, Eq)]
struct Decrement;

struct CounterApp;

impl Widget<CounterState> for CounterApp {
    fn build(&self, ctx: &mut BuildCtx<CounterState>, view: &View<CounterState>) -> Node {
        Column {
            children: vec![
                Text {
                    content: TextContent::Literal(format!("Count: {}", view.state.count)),
                    font_size: Some(24.0),
                    ..Default::default()
                }
                .into(),
                Button {
                    on_press: Some(ctx.bind(Increment, |state, _, _| {
                        state.count += 1;
                    })),
                    child: Some(Box::new(Text::new("Increment").into())),
                    ..Default::default()
                }
                .into(),
                Button {
                    on_press: Some(ctx.bind(Decrement, |state, _, _| {
                        state.count -= 1;
                    })),
                    child: Some(Box::new(Text::new("Decrement").into())),
                    ..Default::default()
                }
                .into(),
            ],
            ..Default::default()
        }
        .into()
    }
}

fn main() -> anyhow::Result<()> {
    DesktopApp::new(CounterApp).run()
}
```

This example shows three core ideas:

1. state is read inside `build` and never mutated there,
2. UI events map to typed actions,
3. reducers are where state changes happen.

Run locally to verify the cycle.

```bash
cargo run --example counter
```
